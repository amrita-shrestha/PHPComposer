# Demo

<?php

echo "Welcome to composer learning";
// use the factory to create a Faker\Generator instance

echo $faker->name();
// 'Vince Sporer'
echo $faker->email();
// 'walter.sophia@hotmail.com'
echo $faker->text();
// 'Numquam ut mollitia at consequuntur inventore dolorem.'
?>
